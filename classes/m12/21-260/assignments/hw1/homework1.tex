\begin{question}{Section 1.1, Problem 24}
Let $m$ denote the mass of the drug in the patient's body.
\begin{enumerate}[(a)]
\item The drug enters the patient's bloodstream at a rate of
$(5 \, mg/cm^3)(100 \, cm^3/h) = 500 \, mg/h$
and leaves the patient's bloodstream at a rate of $(0.4 \, h\inv)m$, so that
\[
    \mbox{
        \fbox{
            $\displaystyle \frac{dm}{dt} = 500 \, mg/h - (0.4 \, h\inv)m$.
        }
    }
\]

\item Quantity of the drug in the patient's bloodstream reaches equilibrium
when $\frac{dm}{dt} = 0 \, mg/h$, so that, by the \diffeq found in part (a),
$0 = 500 \, mg/h - (0.4 \, h\inv)m$. This happens when
\fbox{$m = 1250 \, mg$.}
\end{enumerate}
\end{question}

\begin{question}{Section 1.3, Problem 18}
Suppose $y = e^{rt}$, for some $r \in \mathbb{R}$. Then,
$\forall i \in \mathbb{R}$, the $i^{th}$ derivative of $y$ with respect to
$t$ is $y^{(i)} = r^ie^{rt}$. Thus, if $y^{(3)} - 3y^{(2)} + 2y^{(1)} = 0$,
then, 
\[r^3e^{rt} - 3r^2e^{rt} + 2re^{rt} = 0,
\quad \mbox{ so that } \quad
r^3 - 3r^2 + 2r = 0\]
(since $e^{rt} \neq 0$). Thus, \fbox{$r \in \{0,1,3\}$.}
\end{question}

\begin{question}{Section 2.1, Problem 16}
Multiplication by $e^{\int 2/t \; dt} = t^2$ gives
\[t^2y + 2ty = \cos t.\]
Thus,
\[t^2y = \int(t^2y + 2ty) \; dt
 = \int \cos t \; dt
 = \sin t + C,\]
for some $C \in \mathbb{R}$. Since $y(\pi) = 0$, $0 = (0 + C)/\pi^2$, so that
$C = 0$, and thus
\[y = \mbox{
    \fbox{
        $\displaystyle \frac{\sin t}{t^2}$.
    }
}\]
\end{question}

\begin{question}{Section 2.2, Problem 6}
Since, for $x \neq 0, y \neq \pm 1$, the equation is equivalent to the
equation
\[\frac{1}{\sqrt{1 - y^2}}y^{\prime} = \frac1x,\]
the \diffeq is seperable, so that, as shown in class,
\[\int \frac{1}{\sqrt{1 - y^2}} \; dy = \int \frac1x = \ln(|x|) + C,\]
for some $C \in \mathbb{R}$. Letting $t := \sin\inv(y)$ gives
\[\int \frac{1}{\sqrt{1 - y^2}} \; dy
 = \int \frac{\cos t}{\sqrt{1 - \sin^2t}} \; dt
 = \int \frac{\cos t}{\cos t} \; dt
 = \int 1 \; dt = t = \sin\inv(y).\]
Thus,
\[y = \mbox{\fbox{$\displaystyle \sin(\ln(|x|) + C)$.}}\]

\end{question}
